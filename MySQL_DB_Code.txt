CREATE DATABASE CyCap;

CREATE TABLE account{
	Username varchar(255) UNIQUE,
	Password varchar(255),
	D_O_C varchar(255),
	Email varchar(255),
	PRIMARY KEY(username)
};

CREATE TABLE matchInfo{
	KD int,
	Wins int,
	Loses int,
	Username varchar(255) FOREIGN KEY REFERENCES account(username)
};
